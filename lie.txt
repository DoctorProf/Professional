fun generatePassword(phrase: String): String {
    val replacements = mapOf(
        "–∞" to "a", "–±" to "b", "–≤" to "v", "–≥" to "g", 
        "–¥" to "d", "–µ" to "e", "—ë" to "e", "–∂" to "zh",
        "–∑" to "z", "–∏" to "i", "–π" to "y", "–∫" to "k",
        "–ª" to "l", "–º" to "m", "–Ω" to "n", "–æ" to "o",
        "–ø" to "p", "—Ä" to "r", "—Å" to "s", "—Ç" to "t",
        "—É" to "u", "—Ñ" to "f", "—Ö" to "h", "—Ü" to "ts",
        "—á" to "4", "—à" to "sh", "—â" to "sch", "—ä" to "",
        "—ã" to "y", "—å" to "", "—ç" to "e", "—é" to "yu",
        "—è" to "ya"
    )

    var password = phrase.lowercase()
    replacements.forEach { (key, value) ->
        password = password.replace(key, value, ignoreCase = true)
    }

    // –î–æ–±–∞–≤–ª–µ–Ω–∏–µ —Å–ø–µ—Ü—Å–∏–º–≤–æ–ª–æ–≤
    password = password.replace(" ", "_")
                     .replace("–æ", "0")
                     .replace("—á", "4")
                     .replace("–º", "M")
    return password
}
